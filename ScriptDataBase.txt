
CREATE TABLE PESSOA (
                HANDLE INTEGER NOT NULL,
                STATUSDATA TIMESTAMP NOT NULL,
                STATUS INTEGER NOT NULL,
                LOGUSUARIOALTERACAO INTEGER NOT NULL,
                LOGDATACADASTRO TIMESTAMP NOT NULL,
                LOGDATAALTERACAO TIMESTAMP NOT NULL,
                LOGUSUARIOCADASTRO INTEGER NOT NULL,
                NOME VARCHAR NOT NULL,
                CNPJCPF VARCHAR NOT NULL,
                OBSERVACAO VARCHAR NOT NULL,
                CONSTRAINT handle_pessoa PRIMARY KEY (HANDLE)
);

CREATE TABLE USUARIO (
                HANDLE INTEGER NOT NULL,
                LOGIN VARCHAR NOT NULL,
                SENHA VARCHAR NOT NULL,
                STATUS INTEGER NOT NULL,
                STATUSDATA VARCHAR NOT NULL,
                LOGUSUARIOCADASTRO INTEGER NOT NULL,
                LOGDATAALTERACAO TIMESTAMP NOT NULL,
                LOGDATACADASTRO TIMESTAMP NOT NULL,
                LOGUSUARIOALTERACAO INTEGER NOT NULL,
                CONSTRAINT handle_usuario PRIMARY KEY (HANDLE)
);

CREATE TABLE VENDA (
                HANDLE INTEGER NOT NULL,
                VALORLIQUIDO NUMERIC(15,2) NOT NULL,
                CLIENTE INTEGER NOT NULL,
                OBSERVACAO VARCHAR,
                DATA TIMESTAMP NOT NULL,
                STATUS INTEGER NOT NULL,
                STATUSDATA TIMESTAMP NOT NULL,
                VENDEDOR INTEGER,
                NUMEROPEDIDO VARCHAR,
                LOGUSUARIOCADASTRO INTEGER NOT NULL,
                LOGDATAALTERACAO TIMESTAMP NOT NULL,
                LOGDATACADASTRO TIMESTAMP NOT NULL,
                LOGUSUARIOALTERACAO INTEGER NOT NULL,
                CONSTRAINT handle_venda PRIMARY KEY (HANDLE)
);


CREATE TABLE PRODUTO (
                HANDLE INTEGER NOT NULL,
                VALOR NUMERIC(15,2) NOT NULL,
                NOME VARCHAR,
                CODIGOAUXILIAR VARCHAR,
                ESTOQUE NUMERIC(15,4) NOT NULL,
                DESCRICAO VARCHAR,
                LOGDATACADASTRO TIMESTAMP NOT NULL,
                LOGDATAALTERACAO TIMESTAMP NOT NULL,
                LOGUSUARIOCADASTRO INTEGER NOT NULL,
                LOGUSUARIOALTERACAO INTEGER NOT NULL,
                CONSTRAINT handle_produto PRIMARY KEY (HANDLE)
);


CREATE TABLE VENDAPRODUTO (
                HANDLE INTEGER NOT NULL,
                OBSERVACAO VARCHAR,
                PRODUTO INTEGER NOT NULL,
                QUANTIDADE NUMERIC(15,4) NOT NULL,
                VALORUNITARIO NUMERIC(15,2) NOT NULL,
                VALORTOTAL NUMERIC(15,2) NOT NULL,
                VENDA INTEGER NOT NULL,
                CONSTRAINT handle_vendaproduto PRIMARY KEY (HANDLE)
);

CREATE TABLE STATUS (
                HANDLE INTEGER NOT NULL,
                NOME VARCHAR,
                CONSTRAINT handle_status PRIMARY KEY (HANDLE)
);
